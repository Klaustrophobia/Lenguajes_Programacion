CREATE DATABASE DUMMY;

USE DUMMY;

CREATE TABLE clientes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(30),
    apellido VARCHAR(30),
    identidad VARCHAR(30),
    correo VARCHAR(50),
    contrasenia VARCHAR(30)
);

CREATE TABLE administradores (
    idadministrador INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(30),
    apellido VARCHAR(30),
    correo VARCHAR(50),
    contrasenia VARCHAR(30)
);
